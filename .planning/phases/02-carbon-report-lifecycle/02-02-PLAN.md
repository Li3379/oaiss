if (request.getReviewResult() == ReportStatusEnum.APPROVED.getCode()) {

    // 1. Credit score bonus (+5 points)

    // 4. Transition to ON_CHAIN(5) per D-05
    report.setStatus(ReportStatusEnum.ON_CHAIN.getCode());
}
