2. Create `src/api/marketPrediction.ts`:

```typescript
import request from './request'
import type { MarketPredictionRequest, MarketPredictionResponse } from '@/types/ai'
import type { ApiResponse } from '@/types/api'

export function getMarketTrend(data: MarketPredictionRequest): Promise<ApiResponse<MarketPredictionResponse>> {
  return request.post('/ai/market/trend', data)
}

export function getMarketPrice(data: MarketPredictionRequest): Promise<ApiResponse<MarketPredictionResponse>> {
  return request.post('/ai/market/price', data)
}

export function getSupplyDemand(data: MarketPredictionRequest): Promise<ApiResponse<MarketPredictionResponse>> {
  return request.post('/ai/market/supply-demand', data)
}
```

```typescript
import request from './request'
import type { EnterpriseInferenceResponse } from '@/types/ai'
import type { ApiResponse } from '@/types/api'

export function getEnterpriseInference(enterpriseId: number): Promise<ApiResponse<EnterpriseInferenceResponse>> {
  return request.get(`/predict/enterprise/${enterpriseId}/inference`)
}
```
